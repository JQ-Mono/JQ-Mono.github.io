C++手撕B+树

#include<vector>

using namespace std;

class BPTreeNode{
public:
    BPTreeNode(bool isLeaf = false):
        isLeaf(isLeaf), next(nullptr), father(nullptr){}
    bool isLeaf; // 是否是叶子结点
    vector<int> keys;
    BPTreeNode* next; // 兄弟结点，当isLeaf为true时有效
    BPTreeNode* father; // 指向父亲的指针，当且仅当根结点的父结点为空
    vector<BPTreeNode*> kids; // 孩子结点
    ~BPTreeNode(){}
};

    /****** 插入结点 ******/
    void keyInsert(int key){
        BPTreeNode* cur = root;

        // 首先向下找到正确插入叶子结点的位置
        while(!cur->isLeaf){
            int ind = 0;
            while(ind < cur->keys.size() && cur->keys[ind] < key) ind++;
            cur = cur->kids[ind];
        }

        auto& ckeys = cur->keys;
        // 先直接插入叶子结点，利用插入排序思想，保证结点内keys有序
        for(int& evkey: ckeys){
            if(evkey == key)
                return ; // key已经存在，插入失败，直接返回
            if(evkey > key)
                swap(evkey, key);
        }
        ckeys.push_back(key);

        // 若叶子结点内keys数量达到限额，此时需要分裂
        if(ckeys.size() == level){
            int mid = level / 2; // 角标中间位置

            // 先创建兄弟结点，并调整好底层链表结构
            BPTreeNode* brother = new BPTreeNode(true);
            brother->keys = {ckeys.begin() + mid, ckeys.end()};
            ckeys.resize(mid);
            brother->next = cur->next;
            cur->next = brother;

            // 再更新父亲结点
            update_father(cur->father, cur, brother, brother->keys[0]);
        }
    }

    /****** 删除结点 ******/
    void keyDelete(int key){
        BPTreeNode* cur = root;
        while(!cur->isLeaf){ // 向下查找直到结点为叶子结点
            int ind = 0;
            while(ind < cur->keys.size() && cur->keys[ind] <= key) ind++;
            cur = cur->kids[ind];
        }

        auto& ckeys = cur->keys;
        // 遍历查找，若找到key则删除
        for(int i = 0; i < ckeys.size(); ++i){
            if(ckeys[i] == key){
                for(int j = i + 1; j < ckeys.size(); ++j)
                    ckeys[j - 1] = ckeys[j];
                ckeys.pop_back();
            }
        }

        // 判断cur所含keys数量是否太少，若是则需要调整树结构
        if(cur != root && ckeys.size() < level / 2){
            auto& fkeys = cur->father->keys;
            auto& fkids = cur->father->kids;

            // 先在cur的father中找到cur所在kids的位置
            for(int i = 0; i < fkids.size(); ++i){
                if(fkids[i] == cur){ // 找到位置i
                    // 询问cur左右兄弟，是否有条件给cur借出一个key
                    /// 左兄弟可以借出一个key
                    if(i > 0 && fkids[i - 1]->keys.size() > level / 2){
                        auto& brother = fkids[i - 1];
                        auto& bkeys = brother->keys;

                        int key = bkeys.back();
                        bkeys.pop_back();
                        fkeys[i - 1] = key; // 注意替换父结点的索引

                        // 接下来向cur插入借来的key
                        for(int& evkey: ckeys){
                            if(evkey > key)
                                swap(evkey, key);
                        }
                        ckeys.push_back(key);
                    }
                    /// 左兄弟不能借出，但右兄弟可以
                    else if(i + 1 < fkids.size() && fkids[i + 1]->keys.size() > level / 2){
                        auto& brother = fkids[i + 1];
                        auto& bkeys = brother->keys;

                        int key = bkeys[0];
                        for(int i = 1; i < bkeys.size(); ++i) bkeys[i - 1] = bkeys[i];
                        bkeys.pop_back();
                        fkeys[i] = bkeys[0]; // 注意替换父结点的索引

                        // 接下来向cur插入借来的key
                        ckeys.push_back(key);
                    }
                    /// 两兄弟都无法借出key，只能选一个兄弟进行合并；由于合并，父结点将减少一个key
                    else {
                        if(i > 0){ // 可以与左兄弟合并
                            auto& brother = fkids[i - 1];
                            auto& bkeys = brother->keys;

                            for(int evkey: ckeys) bkeys.push_back(evkey);
                            brother->next = cur->next; // 重新连接链表
                            delete cur; // 删除当前指针
                            // 更新父结点信息
                            for(int j = i; j < fkeys.size(); ++j) fkeys[j - 1] = fkeys[j];
                            fkeys.pop_back();
                            for(int j = i + 1; j < fkids.size(); ++j) fkids[j - 1] = fkids[j];
                            fkids.pop_back();
                        }
                        else if(i + 1 < fkids.size()){ // 可以与右兄弟合并
                            auto& brother = fkids[i + 1];
                            auto& bkeys = brother->keys;

                            for(int evkey: bkeys) ckeys.push_back(evkey);
                            cur->next = brother->next; // 重新连接链表
                            delete brother; // 删除兄弟指针
                            // 更新父结点信息
                            for(int j = i + 1; j < fkeys.size(); ++j) fkeys[j - 1] = fkeys[j];
                            fkeys.pop_back();
                            for(int j = i + 2; j < fkids.size(); ++j) fkids[j - 1] = fkids[j];
                            fkids.pop_back();
                        }

                        if(fkeys.size() < level / 2)
                            update_father(father);
                    }
                } // if
            } // for
        } // if
    }

    /****** 查询结点 ******/
    bool quary(int key){
        BPTreeNode* cur = root;
        while(!cur->isLeaf){ // 向下查找直到结点为叶子结点
            int ind = 0;
            while(ind < cur->keys.size() && cur->keys[ind] <= key) ind++;
            cur = cur->kids[ind];
        }
        for(int evkey: cur->keys){
            if(evkey == key){
                cout<<"已找到key："<<key<<endl;
                return true;
            }
        }
        cout<<"未找到key："<<key<<endl;
        return false;
    }
private:
    int level = 5; // B+树的阶数
    BPTreeNode* root;

    /****** 插入非叶子结点，为递归函数，对外不公开 ******/
    void update_father(BPTreeNode* father, BPTreeNode* left, BPTreeNode* right, int key){
        if(father == nullptr){ // 若当前father为空，需要创建新结点并更新root
            father = new BPTreeNode();
            father->keys.push_back(key);
            father->kids.push_back(left);
            father->kids.push_back(right);
            root = father;
            left->father = father;
            right->father = father;
            return ;
        }

        right->father = father; // 踩了坑，要记得更新
        auto& fkeys = father->keys;
        auto& fkids = father->kids;

        // 在father中插入right结点，并同时将key插入keys对应位置中
        for(int i = 0; i < fkids.size(); ++i){
            if(fkids[i] == left){
                for(int j = i + 1; j < fkids.size(); ++j)
                    swap(fkids[j], right);
                fkids.push_back(right);

                for(int j = i; j < fkeys.size(); ++j)
                    swap(fkeys[j], key);
                fkeys.push_back(key);
                break;
            }
        }

        // 当该父结点达到限额时，需要递归向上更新父结点
        // 相比于叶子结点来说，此时不需要更新next指针
        if(fkeys.size() == level){
            // 分裂，首先创建兄弟结点
            int mid = level / 2;
            int key = fkeys[mid];
            BPTreeNode* brother = new BPTreeNode();
            brother->keys = {fkeys.begin() + mid + 1, fkeys.end()};
            brother->kids = {fkids.begin() + mid + 1, fkids.end()};
            // 注意更新brother结点中孩子们的父结点信息
            for(auto kid: brother->kids) kid->father = brother;
            fkeys.resize(mid);
            fkids.resize(mid + 1);
            // 接下来递归调用
            update_father(father->father, father, brother, key);
        }
    }
