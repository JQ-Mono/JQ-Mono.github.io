c++位运算函数之（__builtin_）

__builtin_系列：

// 判断n的二进制数中有多少个1
int __builtin_popcount(unsigned int n);

// 判断n的二进制数中1的个数的奇偶性
int __builtin_parity(unsigned int n);

int __builtin_ffs(unsigned int n);

// 判断n的二进制数末尾0的个数
int __builtin_ctz(unsigned int n);

// 判断n的二进制数前导0的个数
