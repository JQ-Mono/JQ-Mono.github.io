强连通分量——Tarjan算法

2020-03-15 经典基础算法 416 字
图论
  • 时间复杂度为$O(N+M)$
#include<iostream>
#include<vector>
#define M 10010
using namespace std;
int dfn[M]; // dfn保存每个点的时间戳编号
int low[M]; // low是某个点或其子树能够追溯到最早的栈中结点编号
int Bcnt; // Bcnt记录强连通分量个数
int index; // index是时间戳
int stack[M];
int top; // top是栈顶编号
bool instack[M]; // 判断某个点是否在栈内
vector<int> g[M]; // 邻接表保存图
void tarjan(int i){
	int j;
	dfn[i] = low[i] = ++index; // dfn、low初始化为时间戳
	instack[i] = true; // i入栈，作为一颗（子）dfs树的根
	stack[++top] = i;
	for(int k = 0; k < g[i].size(); ++k){ // 访问以i为起点的所有边
		j = g[i][k]; // j记录以i为起点的某条边的终点
		if(!dfn[j]){ // 若这个点还没被访问，就递归调用函数，压栈
			tarjan(j);
			if(low[j] < low[i])
                low[i] = low[j]; // 若j的根被放小，则把j的根i也放小
		}
        /* 或者j这个点已被访问但是j在栈中，且j的时间戳早于i，将i的low放小 */
		else if(instack[j] && dfn[j] < low[i]) 
            low[i] = dfn[j];
	}
	if(dfn[i] == low[i]){ // i这个点未被放小，则以i为根的一颗dfs子树是一个强连通分量
		Bcnt++;
        /* 弹栈至i被弹出 */
		do{
			j = stack[top--];
			instack[j] = false;
		} while(j != i);
	}
}
/* 使用样例：输入点、边数和所有边，输出强连通分量个数 */
int main()
{
	int n,m;
    cin>>n>>m;
    for(int i = 0; i < m; ++i){
        int s,t;
        cin>>s>>t;
        g[s].push_back(t);
    }
    for(int i = 1; i <= n; ++i){
    	if(!dfn[i]) tarjan(i);
	}
	cout<<Bcnt;
	return 0;
}
