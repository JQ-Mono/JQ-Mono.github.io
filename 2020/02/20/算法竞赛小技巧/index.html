一、快读

/* 读入单个字符比读入整数的效率高 */
/* 函数isdigit():判断字符是否为0-9,头文件#include<cctype> */
//无负数
inline int read(){
    int ans = 0;
    char c = getchar();
    while(!isdigit(c)) c = getchar();
    //左移比加法运算更高效，ans*10=ans*8+ans*2=ans*2^3+ans*2;
    while(isdigit(c)) {
        ans = (ans<<3) + (ans<<1) + c -'0';
        c = getchar();
    }
    return ans;
}
//有负数
inline int read(){
    register int ans = 0, c = getchar(), flag = true;
    while(!isdigit(c)){ // 字符c不是数字，若是负号
        flag = c=='-'? -1 : 1;
        c = getchar();
    }
    while(isdigit(c)){
        ans = ans*10 + c - '0';
        c = getchar();
    }
    return ans * flag;
}
// string类快读
inline string read_str(){
	char ch = getchar();
	string str;
	while (ch == ' ' || ch == '\n') ch = getchar(); // 删去前面无用字符
	while(ch != ' ' && ch != '\n') str += ch, ch = getchar();
	return str;
}

二、快速幂

/* 计算x的n次方模mod的结果 */
int pow(int x,int n){
    int ans = 1;
    while(n){
        if(n & 1) ans *= x;
        x *= x;
        n >>= 1;
    }
    return ans;
}
/* 以下是高精度取模版本 */
typedef long long ll;
#define mod 1000000007
ll pow(ll x, ll n){
    ll ans = 1;
    while(n){
        if(n & 1) ans = (ans * x) % mod;
        x = (x * x) % mod;
        n >>= 1;
    }
    return ans;
}

三、组合数的计算

long long C(long long n, long long m){
    long long ans = 1;
    for(long long i = 1; i <= m; ++i){
        ans = ans * (n - m + i) / i; // 一定要先乘再除
    }
    return ans;
}

五、最大公约数和最小公倍数

/* 最大公约数 */
int gcd(int a, int b){
	int ans; // 保存最大公约数 
	/* 根据两数最小确定循环次数 */ 
	int min = a < b ? a : b;
	for(int i = 1; i <= min; i++)
        if(a % i == 0 && b % i == 0)
			ans = i;
	return ans;
}
/* 最大公约数递归写法，复杂度为O(log(max(x,y))) */
int gcd(int x, int y){
    return y ? gcd(y, x % y) : x;
}
/* 最小公倍数 */
int lcm(int x, int y){
    return x / gys(x, y) * y; // 一定程度上防止溢出
}

六、简单字符串分割实现

vector<string> split(string str, char by){
    vector<string> ans;
    string ev;
    for(int i = 0; i < str.size(); ++i){
        if(str[i] != by){
            ev += str[i];
        } else {
            if(ev.size()) ans.push_back(ev);
            ev.clear();
        }
    }
    if(ev.size()) ans.push_back(ev);
    return ans;
}
